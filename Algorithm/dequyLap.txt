#include <iostream>
using namespace std;
int fibo(int n)
{
	int a = 0;
	int b = 1;
	int next = 0;
	for (int i = 0; i <= n; i++)
	{
		if (i <= 1) next = i;
		else {
			next = a + b;
			a = b;
			b = next;
		}
	}
	return next;
}
int main(void)
{
	for (int i = 0; i <= 10; i++) {
		cout << fibo(i) << ", ";
	}
	system("pause");
}
