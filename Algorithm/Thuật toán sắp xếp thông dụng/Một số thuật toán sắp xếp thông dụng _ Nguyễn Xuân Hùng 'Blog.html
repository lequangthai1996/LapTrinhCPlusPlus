Một số thuật toán sắp xếp thông dụng

Khói 7:11:00 AM Lập Trình C , Thuật Toán Edit
Mình xin chia sẻ với các bạn một số thuật toán sắp xếp 
như bubble sort ( sắp xếp nổi bọt ) , insertion sort (sắp xếp chèn), selection sort (sắp xếp lựa chọn), quick sort (sắp xếp nhanh), merge sort 
Mình sử dụng chương trình dịch là DEV C++ 4.9

Bạn nào biết rồi thì có thể xem thêm ,có gì góp ý cho mình.



1/ Insertion sort 
Sở dĩ nó có tên như vậy vì mỗi lần lặp lại của vòng lặp chính nó chèn phần tử kế tiếp vào đúng vị trí của mảng đang xử lý . 

void insertion_sort(int A[],int n) 
{
     for(int i=2;i<=n;i++)
     {
             x=A[i];j=i-1; 
             while((x<A[j])&&(j>0))
             {
                         A[j+1]=A[j];
                         j--;
             }
             A[j+1]=x;
     }
     
}
2/ Selection sort
sắp xếp kiểu chọn làm việc giống như bubble sort , mỗi lần lặp lại của vòng lặp i chính phần tử lớn nhất kế được di chuyển đúng vào vị trí của nó , nhưng thay vì dịch chuyển những phần tử này vào đúng vị trí , thuật toán tìm kiếm phần tử cần được di chuyển trước bất kỳ phần tử nào, sau đó đặt đúng vị trí bằng sự tráo đổi

void selection_sort(int A[],int n) 
{
     for(int i=1;i<n;i++)
     {
             m=i;
             for(int j=i+1;j<=n;j++)
             if(A[j]<A[m]) m=j;
             if(m!=i) swap(A[i],A[m]);
     }
     
}
void swap(int &x,int &y)
{
     tmp=x;
     x=y;
     y=tmp;
}
3/ Bubble sort
Đây có thể coi là thuật toán đơn giản nhất trong số những thuật toán sắp xếp
Cơ bản ý tưởng của thuật toán này như là " đun nước sôi "
các bong bóng nhẹ hơn sẽ nổi lên trên 


void bubble_sort(int A[],int n) {
     for(int i=1;i<=n-1;i++)
     {
             
             for(int j=n;j>i;j--)
             {
                     if(A[j]<A[j-1])
                     {
                     swap(A[j],A[j-1]);
                     }
             }
     }
     
}
void swap(int &x,int &y)
{
     tmp=x;
     x=y;
     y=tmp;
}
4/ Quick sort
mình thích nhất cái này .Thuật toán này chia dãy thành 2 phần tách biệt nhau bởi 1 phần tử pivot , dãy 1 gồm các phần tử nhỏ hơn pivot , dãy 2 gồm các phần tử lớn hơn pivot ,với mục tiêu hoán đổi vị trí các phần tử nhỏ hơn pivot được chuyển sang bên trái , các phần tử lớn hơn sang bên phải,

5/ Merge sort ( sắp xếp trộn hoặc sắp xếp hòa nhập ) 
Đây cũng có thể coi là một thuật toán sắp xếp khó 

// sap xep tron

void m_sort(int numbers[], int temp[], int left, int right)
{
  int mid; // mid la vi tri giua cua mang numbers khi bi chia doi 
 
  if (right > left) // left la phan tu dau cua nua trai
  {                 // right la phan tu cuoi cua nua phai
    mid = (right + left) / 2;
    
    m_sort(numbers, temp, left, mid); // De quy sap xep tron nua trai
    m_sort(numbers, temp, mid+1, right); // De quy sap xep tron nua phai 
    merge(numbers, temp, left, mid+1, right);
  }
}
 
void merge(int numbers[], int temp[], int left, int mid, int right)// tron
{
  int i, left_end, num_elements, tmp_pos;
 
  left_end = mid - 1; // phan tu cuoi cua nua trai 
  tmp_pos = left;
  num_elements = right - left + 1;
 
 // [left ....left_end]   [    .... right]
  while ((left <= left_end) && (mid <= right))
  {
    if (numbers[left] <= numbers[mid])
    {
      temp[tmp_pos] = numbers[left];
      tmp_pos = tmp_pos + 1;
      left = left +1;
    }
    else
    {
      temp[tmp_pos] = numbers[mid];
      tmp_pos = tmp_pos + 1;
      mid = mid + 1;
    }
  }
  while (left <= left_end) // sao chep nua trai 
  { 
    temp[tmp_pos] = numbers[left];
    left = left + 1;
    tmp_pos = tmp_pos + 1;
  }
  while (mid <= right) //sao chep nua phai 
  {
    temp[tmp_pos] = numbers[mid];
    mid = mid + 1;
    tmp_pos = tmp_pos + 1;
  }
 
  for (i=0; i <= num_elements; i++) // sao chep 
  {
    numbers[right] = temp[right];
    right = right - 1;
  }
}
 6. shaker sort :

void ShakerSort(int a[],int n)
{
 int l = 0 ;
 int r = n - 1; // lay phan tu gan cuoi 
 int j_last = n - 1; // luu chi so lan so sanh cuoi cung
 while(l < r) // lam den khi l > r || l == r
 {
  for ( int j = r ; j > l ; j--)
  {
   if ( a[j] < a[j-1])
   {
    swap(a[j-1],a[j]);
    j_last = j; // luu chi so cua j vao j_last
   }
  }
  l = j_last; // lay gia tri chi so cua j_last dua vao l
  for ( int j = l; j < r; j++)
  {
   if ( a[j] > a[j+1])
   {
    swap(a[j],a[j+1]);
    j_last = j; // luu gai tri chi so
   }
  }
  r = j_last; // luu gia tri chi so ra "r" co gia tri moi 
}

 8. heap sort :

void Heapsort(int a[], int n) 
{
    int k,x,s,f,ivalue;
    for(k=1;k<n;k++){
        x=a[k];
        s=k; f=(s-1)/2;
        while(s>0 && a[f]<x){
            a[s]=a[f];
            s=f; f=(s-1)/2;
        }
        a[s]=x;
    }
    for(k=n-1;k>0;k--){
        ivalue=a[k];
        a[k]=a[0];
        f=0;
        if(k==1)
            s=-1;
        else
            s=1;
        if(k>2 && a[2]>a[1])
            s=2;
        while(s>=0 && ivalue<a[s]){
            a[f]=a[s];
            f=s;s=2*f +1;
            if (s+1<=k-1 && a[s]<a[s+1])
                s=s+1;
            if (s>k-1)
                s=-1;
        }
        a[f]=ivalue;
    }
}

 }

  • 10. Binary Insertion Sort:

Binary Insertion Sort là cải tiến của Insertion Sort.

void Binary_Insertion_Sort(int a[], int n)
{
    int left, right, m , i;
    int key; //lưu giá trị Array[i] tránh bị ghi đè khi dời vị trí
    for (i = 1; i < n; i++)
    {
        key = a[i];
        left = 0;
        right = i - 1;
        while (left <= right)
        {
            m = (left + right) / 2; //tìm vị trí thích hợp của m
            if (key < a[m])
                right = m - 1;
            else left = m + 1;
        }
        for (int j = i - 1; j >= left; j--)
            a[j+1] = a[j]; //dời các ptử đứng sau key
        a[left] = key; //chèn key vào đây
    }
}

  • 11/ Gnome sort

    Gnome sort là một trong nhữngthuật toán sắp xếp đơn giản nhất, thuộc nhóm các thuật toán có độ phức tạp là O(n2) như sắp xếp nổi bọt, sắp xếp chèn,…. tuy nhiên điểm khác biệt là của Gnome sort là chúng ta chỉ cần sử dụng 1 vòng lặp thay vì 2 vòng lặp so với các thuật toán khác trong cùng nhóm.

    Ý tưởng của Gnome thật đơn giản, bắt đầu đi từ vị trí thứ 2 của dãy giá trị cần sắp xếp tăng dần, so sánh giá trị phần tử ở vị trí hiện tại với phần tử phía trước nó. Nếu nhỏ hơn thì hoán vị rồi lùi lại một vị trí (giới hạn biên là vị trí xuất phát), nếu lớn hơn thì tiến lên một vị trí. Cứ như vậy cho tới khi tiến tới vị trí cuối cùng của dãy giá trị.

    Thuật toán:

    Bước 1:
    Gán i = 1; // duyệt từ vị trí thứ 2
    Gán j = 2; // theo dõi các vị trí kế tiếp, sau khi kết thúc vòng lặp hiện tại thì nhảy tới vị trí này để tiếp tục lặp
    Bước 2:
    Nếu (Array[i] < Array[i-1]) thì {Swap(Array[i], Array[i-1]); –i}
    Nếu (i==0) thì i = j++
    Ngược lại i = j++;
    Bước 3:
    Nếu i < n thì lặp lại bước 2
    Ngược lại: dừng

    12. COUNTING SORT


    Counting Sort là thuật toán đếm số lần xuất hiện của các giá trị khác nhau trong mảng ban đầu. Và gán giá trị vừa đếm vào vị trí tương ứng của 1 mảng có độ dài là giá trị lớn nhất của mảng.

    Giải thuật:


    Bước 1: Tìm MAX = max(Array[]);
    Bước 2: Tạo mảng động *Soluong có độ dài = MAX và Soluong[i] = 0;
    int *Soluong = new (int *)calloc(MAX, sizeof(int));
    Bước 3: Đếm số lần xuất hiện của các giá trị khác nhau trong mảng ban đầu. Và gán giá trị vào vị trí tương ứng ở mảng Soluong.
    Bước 4: Sắp xếp lại mảng ban đầu bằng cách duyệt mảng Soluong.

long long LONNHAT(int *Array, int size)
{
    long long MAX = Array[0];
    for (int i = 0; i < size; i++)
        if (Array[i] > MAX)
            MAX = Array[i];
    return MAX;
}
void countingSort(int *Array, int size)
{
    long long MAX = LONNHAT(Array,size);
    int *soluong = (int *)calloc(MAX, sizeof(int));
    for (int i = 0; i < size; i++)
        soluong[Array[i]]++;
//sắp xếp lại mảng ban đầu
    int outputindex = 0;
    for (int i = 0; i < MAX; i++)
      while (soluong[i]--)
        {
            Array[outputindex++] = i;
        }
    delete []soluong; //xóa mảng động
}
