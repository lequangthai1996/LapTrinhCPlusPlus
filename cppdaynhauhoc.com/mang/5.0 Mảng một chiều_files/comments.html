// in ra mảng a
for (size_t i = 0; i < a.size(); i++)
    std::cout << a[i] << " ";

a.clear(); //a = []
a.push_back(1); //a = [1]
a.push_back(2); //a = [1,2]
a.push_back(3); //a = [1,2,3]
a.push_back(4); //a = [1,2,3,4]
a.push_back(5); //a = [1,2,3,4,5], truy cập được a[0..4]

std::vector<int> b(5, 0); //b = [0,0,0,0,0]

b.resize(2); //b = [0,0]

int vitri_max[30];
int vitri_max_count = 0; //cho biết có bao nhiêu học sinh có cùng điểm số cao nhất

code cần sửa/thêm vài dòng:

  • khi diem_max == mang_diem[i] thì cần thêm i vào vitri_max:

vitri_max[vitri_max_count] = i;
vitri_max_count++; //mỗi lần thêm i vào thì vitri_max_count phải tăng 1
  • khi diem_max < mang_diem[i], ví dụ vitri_max đang có 3 hs 8 điểm, thì vitri_max_count = 3, bây giờ xuất hiện 1 hs 9 điểm, như vậy vitri_max chỉ còn 1 hs 9 điểm, vậy vitri_max_count phải reset lại:

vitri_max[0] = i; //vị trí của hs đầu tiên
vitri_max_count = 1; //reset lại

kết quả trong vòng for là

if (diem_max == mang_diem[i])
