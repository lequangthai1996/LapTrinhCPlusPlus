Đúng vậy, em dùng array::fill đi kết hợp thêm với std::fill_n cho có nhiều option.

http://www.cplusplus.com/reference/array/array/fill/
http://www.cplusplus.com/reference/algorithm/fill_n/

#include <iostream>
#include <array>
#include <algorithm>

int main() {
        std::array<int,6> dat;
        dat.fill(0);
        std::fill_n(dat.begin(), 3, 4);
        for(const auto & v : dat)
                std::cout << v << " ";
}
