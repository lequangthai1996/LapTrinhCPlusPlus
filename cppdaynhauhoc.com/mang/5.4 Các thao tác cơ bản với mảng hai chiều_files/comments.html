//Tong Cac Phan Tu duong cheo
using namespace std;

int main()
{
    cout << "Mang Hinh Vuong Co Kich Thuoc:  " << endl;
    int32_t m;
    cin >> m;
    int32_t arr[m][m];

    //Nhap cac phan tu trong mang;
    for(int i= 0; i<m;i++ )
    {
        for(int t=0; t <m;t++)
        {
            cin >> arr[i][t];
        }
    }

    //Tinh tong cac phan tu nam tren duong cheo
    int tong=0;
    for(int n = 0; n<m;n++)
    {
        tong += arr[n][n];
    }
    cout << tong<< endl;
    return 0;
}
