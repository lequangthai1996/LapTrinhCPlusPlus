// Giải phương trình bậc nhất ax + b = 0

#include iostream;
using namespace std;

int a, b ;

int main {
    cout << "Nhap vao 2 so a va b:" << endl;
    cin >> a >> b ;
    cout << " Phuong trinh co nghiem x = " << -b/a << endl;
    system("pause");
return 0;
}

P/s: sẽ có 3 trường hợp xảy ra
TH1 : a = 0 && b = 0 >> phương trình có vô số nghiệm
TH2 : a = 0 && b # 0 >> phương trình vô nghiệm
TH3 : a # 0 && b # 0 như lời giải

Đợi học tới câu lệnh điều kiện thì giải lại bài này chuẩn hơn
